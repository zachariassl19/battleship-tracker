<script>
const SUPABASE_URL = "https://lrjacitfxbntrwmwytpc.supabase.co";
const SUPABASE_KEY = "sb_publishable_5xOMVJlZnqF5IfAUxKUQEQ_dYccpJNv";
const TABLE = "battleship_shots";
const ROWS = "ABCDEFGHIJKLMNO".split("");
const COLS = Array.from({ length: 15 }, (_, index) => index + 1);
const COMMANDS = {
BT: { team: "blue", result: "hit" },
BB: { team: "blue", result: "miss" },
RT: { team: "red", result: "hit" },
RB: { team: "red", result: "miss" }
};
const TEAM_LABEL = { blue: "Blå", red: "Rød" };
const RESULT_LABEL = { hit: "Treff", miss: "Bom" };
const clientId = localStorage.getItem("battleshipClientId") || crypto.randomUUID();
localStorage.setItem("battleshipClientId", clientId);

const supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
let shots = [];
let highlightedCoordinate = "";

const elements = {
status: document.getElementById("connectionStatus"),
commandInput: document.getElementById("commandInput"),
saveCommand: document.getElementById("saveCommand"),
undoButton: document.getElementById("undoButton"),
exportButton: document.getElementById("exportButton"),
copyButton: document.getElementById("copyButton"),
resetButton: document.getElementById("resetButton"),
refreshButton: document.getElementById("refreshButton"),
searchInput: document.getElementById("searchInput"),
message: document.getElementById("message"),
stats: document.getElementById("stats"),
blueBoard: document.getElementById("blueBoard"),
redBoard: document.getElementById("redBoard"),
blueBoardCount: document.getElementById("blueBoardCount"),
redBoardCount: document.getElementById("redBoardCount"),
logFilter: document.getElementById("logFilter"),
logList: document.getElementById("logList"),
jsonBox: document.getElementById("jsonBox"),
importButton: document.getElementById("importButton"),
clearJsonButton: document.getElementById("clearJsonButton")
};

function normalizeCoordinate(value) {
const match = String(value || "").trim().toUpperCase().match(/^([A-O])\s*(1[0-5]|[1-9])$/);
if (!match) return null;
return {
coordinate: `${match[1]}${Number(match[2])}`,
row_label: match[1],
col_number: Number(match[2])
};
}

function parseCommand(value) {
const parts = String(value || "").trim().toUpperCase().split(/\s+/);
if (parts.length !== 2 || !COMMANDS[parts[0]]) return null;
const coordinate = normalizeCoordinate(parts[1]);
if (!coordinate) return null;
return { ...COMMANDS[parts[0]], ...coordinate, client_id: clientId };
}

function setStatus(text, mode = "") {
elements.status.className = `status ${mode}`.trim();
elements.status.querySelector("span:last-child").textContent = text;
}

function showMessage(text, type = "") {
elements.message.textContent = text;
elements.message.className = `message ${type}`.trim();
}

function formatTime(value) {
return new Intl.DateTimeFormat("no-NO", {
hour: "2-digit",
minute: "2-digit",
second: "2-digit",
day: "2-digit",
month: "2-digit"
}).format(new Date(value));
}

function getShotMap(team) {
return new Map(shots.filter((shot) => shot.team === team).map((shot) => [shot.coordinate, shot]));
}

function renderBoard(team, target) {
const map = getShotMap(team);
target.innerHTML = "";
target.appendChild(cell("", "header"));
COLS.forEach((col) => target.appendChild(cell(col, "header")));
ROWS.forEach((row) => {
target.appendChild(cell(row, "header"));
COLS.forEach((col) => {
const coord = `${row}${col}`;
const shot = map.get(coord);
const classes = ["cell"];
if (shot) classes.push("shot", shot.result);
if (highlightedCoordinate === coord) classes.push("highlight");
target.appendChild(cell(shot ? (shot.result === "hit" ? "X" : "•") : "", classes.join(" "), coord));
});
});
}

function cell(text, className, title = "") {
const node = document.createElement("div");
node.className = className;
node.textContent = text;
if (title) node.title = title;
return node;
}

function calculateStats(team) {
const scoped = team === "all" ? shots : shots.filter((shot) => shot.team === team);
const hits = scoped.filter((shot) => shot.result === "hit").length;
const misses = scoped.filter((shot) => shot.result === "miss").length;
const total = scoped.length;
const percent = total ? Math.round((hits / total) * 1000) / 10 : 0;
return { hits, misses, total, percent };
}

function renderStats() {
const cards = [
["Blått lag", calculateStats("blue")],
["Rødt lag", calculateStats("red")],
["Samlet", calculateStats("all")]
];
elements.stats.innerHTML = cards.map(([title, stat]) => `
<div class="stat-card">
<div class="stat-title">${title}</div>
<div class="stat-main">${stat.percent}%</div>
<div class="stat-line">${stat.hits} treff / ${stat.misses} bom</div>
<div class="stat-line">${stat.total} skudd totalt</div>
</div>
`).join("");
elements.blueBoardCount.textContent = `${calculateStats("blue").total} skudd`;
elements.redBoardCount.textContent = `${calculateStats("red").total} skudd`;
}

function renderLog() {
const filter = elements.logFilter.value;
const filtered = shots
.filter((shot) => filter === "all" || shot.team === filter)
.slice()
.sort((a, b) => new Date(b.created_at) - new Date(a.created_at))
.slice(0, 50);

if (!filtered.length) {
elements.logList.innerHTML = `<div class="stat-line">Ingen skudd å vise.</div>`;
return;
}

elements.logList.innerHTML = filtered.map((shot) => `
<div class="log-item">
<span class="badge ${shot.team}">${TEAM_LABEL[shot.team]}</span>
<strong>${shot.coordinate}</strong>
<span class="${shot.result === "hit" ? "result-hit" : "result-miss"}">${RESULT_LABEL[shot.result]}</span>
<span class="time">${formatTime(shot.created_at)}</span>
</div>
`).join("");
}

function renderAll() {
renderBoard("blue", elements.blueBoard);
renderBoard("red", elements.redBoard);
renderStats();
renderLog();
}

async function loadShots() {
const { data, error } = await supabase
.from(TABLE)
.select("*")
.order("created_at", { ascending: true });

if (error) {
setStatus("Feil ved lasting", "error");
showMessage(error.message, "bad");
return;
}

shots = data || [];
renderAll();
setStatus("Sanntid aktiv", "online");
}

async function saveCommand() {
const parsed = parseCommand(elements.commandInput.value);
if (!parsed) {
showMessage("Ugyldig kommando. Bruk for eksempel BT A1 eller RB O15.", "bad");
return;
}

const duplicate = shots.some((shot) => shot.team === parsed.team && shot.coordinate === parsed.coordinate);
if (duplicate) {
showMessage(`${TEAM_LABEL[parsed.team]} har allerede registrert ${parsed.coordinate}.`, "bad");
return;
}

const { error } = await supabase.from(TABLE).insert(parsed);
if (error) {
showMessage(error.code === "23505" ? "Denne ruten er allerede registrert for laget." : error.message, "bad");
await loadShots();
return;
}

elements.commandInput.value = "";
showMessage("Skudd registrert og lagret.", "good");
await loadShots();
}

async function undoLastShot() {
const latest = shots.slice().sort((a, b) => new Date(b.created_at) - new Date(a.created_at))[0];
if (!latest) {
showMessage("Ingen skudd å angre.", "bad");
return;
}

const ok = confirm(`Angre siste skudd: ${TEAM_LABEL[latest.team]} ${latest.coordinate} ${RESULT_LABEL[latest.result].toLowerCase()}?`);
if (!ok) return;

const { error } = await supabase.from(TABLE).delete().eq("id", latest.id);
if (error) {
showMessage(error.message, "bad");
return;
}

showMessage("Siste skudd er fjernet.", "good");
await loadShots();
}

async function resetBoard() {
const ok = confirm("Er du sikker på at du vil nullstille hele brettet? Dette sletter alle skudd.");
if (!ok) return;

const { error } = await supabase.from(TABLE).delete().neq("id", "00000000-0000-0000-0000-000000000000");
if (error) {
showMessage(error.message, "bad");
return;
}

showMessage("Brettet er nullstilt.", "good");
await loadShots();
}

function exportJson() {
const payload = {
exported_at: new Date().toISOString(),
shots: shots.map(({ id, team, result, coordinate, row_label, col_number, created_at }) => ({
id, team, result, coordinate, row_label, col_number, created_at
}))
};
elements.jsonBox.value = JSON.stringify(payload, null, 2);
showMessage("JSON eksportert til tekstfeltet.", "good");
}

async function copyJson() {
if (!elements.jsonBox.value.trim()) exportJson();
await navigator.clipboard.writeText(elements.jsonBox.value);
showMessage("JSON kopiert.", "good");
}

function normalizeImportedShot(raw) {
const coord = normalizeCoordinate(raw.coordinate);
const team = String(raw.team || "").toLowerCase();
const result = String(raw.result || "").toLowerCase();
if (!coord || !["blue", "red"].includes(team) || !["hit", "miss"].includes(result)) return null;
return { team, result, ...coord, client_id: clientId };
}

async function importJson() {
let parsed;
try {
parsed = JSON.parse(elements.jsonBox.value);
} catch {
showMessage("JSON kunne ikke leses.", "bad");
return;
}

const inputShots = Array.isArray(parsed) ? parsed : parsed.shots;
if (!Array.isArray(inputShots)) {
showMessage("JSON må være en liste eller ha feltet shots.", "bad");
return;
}

const normalized = inputShots.map(normalizeImportedShot).filter(Boolean);
if (!normalized.length) {
showMessage("Fant ingen gyldige skudd å importere.", "bad");
return;
}

const unique = Array.from(new Map(normalized.map((shot) => [`${shot.team}:${shot.coordinate}`, shot])).values());
const { error } = await supabase.from(TABLE).upsert(unique, {
onConflict: "team,coordinate",
ignoreDuplicates: true
});

if (error) {
showMessage(error.message, "bad");
return;
}

showMessage(`${unique.length} skudd importert. Duplikater ble hoppet over.`, "good");
await loadShots();
}

function applySearch() {
const coord = normalizeCoordinate(elements.searchInput.value);
highlightedCoordinate = coord ? coord.coordinate : "";
renderAll();
if (elements.searchInput.value.trim() && !coord) {
showMessage("Søk må være en rute fra A1 til O15.", "bad");
}
}

function setupRealtime() {
supabase
.channel("battleship-tracker")
.on("postgres_changes", { event: "*", schema: "public", table: TABLE }, () => {
loadShots();
})
.subscribe((status) => {
if (status === "SUBSCRIBED") setStatus("Sanntid aktiv", "online");
if (status === "CHANNEL_ERROR" || status === "TIMED_OUT") setStatus("Sanntid frakoblet", "error");
});
}

elements.saveCommand.addEventListener("click", saveCommand);
elements.commandInput.addEventListener("keydown", (event) => {
if (event.key === "Enter") saveCommand();
});
elements.undoButton.addEventListener("click", undoLastShot);
elements.resetButton.addEventListener("click", resetBoard);
elements.refreshButton.addEventListener("click", loadShots);
elements.exportButton.addEventListener("click", exportJson);
elements.copyButton.addEventListener("click", copyJson);
elements.importButton.addEventListener("click", importJson);
elements.clearJsonButton.addEventListener("click", () => { elements.jsonBox.value = ""; });
elements.searchInput.addEventListener("input", applySearch);
elements.logFilter.addEventListener("change", renderLog);

setupRealtime();
loadShots();
